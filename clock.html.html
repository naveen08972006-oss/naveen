<script>

/*

  Modern Clock

  - Uses Intl.DateTimeFormat().formatToParts(date, { timeZone })

  - Updates with requestAnimationFrame for smoothness

*/

const FACE = document.getElementById('face');

const hourHand = document.getElementById('hourHand');

const minHand = document.getElementById('minHand');

const secHand = document.getElementById('secHand');

const digital = document.getElementById('digital');

const dateEl = document.getElementById('date');

const tzSelect = document.getElementById('tz');

const tzList = document.getElementById('tzList');

const themeToggle = document.getElementById('themeToggle');

const app = document.getElementById('app').closest('.card') || document.getElementById('app');

// fill ticks (12 hr marks + minute marks)

function buildTicks(){

  const face = FACE;

  face.innerHTML = ''; // clear (hands will be recreated below)

  for(let i=0;i<60;i++){

    const t = document.createElement('div');

    const isHour = i%5===0;

    t.className = 'tick' + (isHour ? '' : ' min');

    t.style.transform = `translate(-50%,0) rotate(${i*6}deg)`;

    face.appendChild(t);

  }

  // re-add hands and cap

  face.appendChild(hourHand);

  face.appendChild(minHand);

  face.appendChild(secHand);

  const cap = document.createElement('div');

  cap.className = 'hand-tail';

  face.appendChild(cap);

}

buildTicks();

// known quick zones for buttons

const quickZones = [

  { id:'local', label: 'Local' },

  { id:'UTC', label:'UTC' },

  { id:'Asia/Kolkata', label:'IST (Asia/Kolkata)' },

  { id:'America/Los_Angeles', label:'PST/PDT (Los Angeles)' },

  { id:'America/New_York', label:'EST/EDT (New York)' },

  { id:'Europe/Berlin', label:'CET/CEST (Berlin)' },

  { id:'Asia/Tokyo', label:'JST (Tokyo)' }

];

function buildTZButtons(){

  quickZones.forEach(z=>{

    const b = document.createElement('button');

    b.className = 'tz-item';

    b.textContent = z.label;

    b.dataset.tz = z.id;

    b.addEventListener('click', ()=> {

      tzSelect.value = z.id;

      currentZone = z.id;

    });

    tzList.appendChild(b);

  });

}

buildTZButtons();

let currentZone = 'local';

// helper: get parts for a timezone using Intl.formatToParts

function getTimePartsForZone(timeZone){

  const now = new Date();

  const opts = { hour: 'numeric', minute:'numeric', second:'numeric', day:'2-digit', month:'short', year:'numeric', hour12:false, timeZone };

  const formatter = new Intl.DateTimeFormat('en-US', opts);

  const parts = formatter.formatToParts(now);

  // formatToParts returns array with types: hour, minute, second, day, month, year, literal...

  const obj = {};

  for(const p of parts){

    if(p.type !== 'literal') obj[p.type] = p.value;

  }

  // ensure numeric numbers

  const h = Number(obj.hour ?? 0);

  const m = Number(obj.minute ?? 0);

  const s = Number(obj.second ?? 0);

  const day = obj.day ?? '';

  const mon = obj.month ?? '';

  const yr = obj.year ?? '';

  return { h, m, s, day, mon, yr };

}

// For "local" we just use native Date

function getLocalParts(){

  const now = new Date();

  return {

    h: now.getHours(),

    m: now.getMinutes(),

    s: now.getSeconds(),

    day: String(now.getDate()).padStart(2,'0'),

    mon: now.toLocaleString('en-US',{month:'short'}),

    yr: String(now.getFullYear())

  };

}

// format digital clock

function format2(n){ return String(n).padStart(2,'0'); }

// main update loop (uses requestAnimationFrame for smoothness)

let lastTimestamp = 0;

function updateLoop(ts){

  // throttle to ~60fps but we only need smooth seconds; still do RAF

  const zone = currentZone;

  let parts;

  if(zone === 'local'){

    // use high-resolution time for smoother seconds (including fractional)

    const now = new Date();

    const ms = now.getMilliseconds();

    parts = {

      ...getLocalParts(),

      frac: ms / 1000

    };

  } else {

    // for timezone-specific hours/minutes/seconds we can still get seconds, but frac will use local ms (small drift possible by <1s)

    const base = getTimePartsForZone(zone);

    parts = { ...base, frac: (new Date()).getMilliseconds()/1000 };

  }

  // compute rotations

  const secondTotal = parts.s + (parts.frac ?? 0);

  const secondDeg = (secondTotal / 60) * 360; // 0-360

  const minuteTotal = parts.m + (secondTotal/60);

  const minuteDeg = (minuteTotal / 60) * 360;

  const hourIn12 = parts.h % 12;

  const hourTotal = hourIn12 + (minuteTotal/60);

  const hourDeg = (hourTotal / 12) * 360;

  hourHand.style.transform = `translate(-50%,-50%) rotate(${hourDeg}deg)`;

  minHand.style.transform = `translate(-50%,-50%) rotate(${minuteDeg}deg)`;

  secHand.style.transform = `translate(-50%,-50%) rotate(${secondDeg}deg)`;

  // digital display uses integer seconds

  const digH = format2(parts.h);

  const digM = format2(parts.m);

  const digS = format2(parts.s);

  digital.textContent = `${digH}:${digM}:${digS}`;

  // date

  dateEl.textContent = `${parts.day} ${parts.mon} ${parts.yr} — ${zone === 'local' ? Intl.DateTimeFormat().resolvedOptions().timeZone : zone}`;

  // loop

  lastTimestamp = ts;

  requestAnimationFrame(updateLoop);

}

// init

requestAnimationFrame(updateLoop);

// hook select

tzSelect.addEventListener('change', (e)=>{

  currentZone = e.target.value;

});

// theme toggle

let light = false;

themeToggle.addEventListener('click', ()=>{

  light = !light;

  if(light){

    document.documentElement.classList.add('light');

  } else {

    document.documentElement.classList.remove('light');

  }

});

// keyboard accessibility: change tz with keys (left/right)

document.addEventListener('keydown', (e)=>{

  if(e.key === 'ArrowLeft' || e.key === 'ArrowRight'){

    const opts = Array.from(tzSelect.options).map(o=>o.value);

    const idx = opts.indexOf(currentZone);

    const delta = e.key === 'ArrowRight' ? 1 : -1;

    const nxt = opts[(idx + delta + opts.length) % opts.length];

    tzSelect.value = nxt;

    currentZone = nxt;

  }

});
